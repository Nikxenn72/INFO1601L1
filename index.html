function getAverageGrade(student, course) { const courseObj = findCourse(student.transcript, course); if (!courseObj || !courseObj.grades || courseObj.grades.length === 0) { return -1; } const sum = courseObj.grades.reduce((acc, grade) => acc + grade, 0); return sum / courseObj.grades.length; } function getAssignmentMark(student, course, num) { const courseObj = findCourse(student.transcript, course); if (!courseObj || !courseObj.grades || num < 1 || num > courseObj.grades.length) { return -1; } // num is 1-based → index = num - 1 return courseObj.grades[num - 1]; } function courseAverageAssessment(students, courseName, assignment) { let total = 0; let count = 0; for (const student of students) { const mark = getAssignmentMark(student, courseName, assignment); if (mark !== -1) { total += mark; count++; } } if (count === 0) { return -1; } return total / count; }
